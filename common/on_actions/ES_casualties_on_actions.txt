on_actions = {
	on_startup = {
		effect = {
			every_country = {
				init_wounded_system = yes
			}
		}
	}

	on_weekly = {
		effect = {
			log = "weekly_effect [THIS.GetName]"
			calculate_healed_wounded = yes
			calculate_wounded_casualties = yes
		}
	}

	on_weekly_RUS = {
		effect = {
			if = {
				limit = { has_war = yes }
				calculate_prisoners_of_war_captured_by_UKR = yes
			}
		}
	}

	on_weekly_UKR = {
		effect = {
			if = {
				limit = { has_war = yes }
				calculate_prisoners_of_war_captured_by_RUS = yes
			}
		}
	}
}
