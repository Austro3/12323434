country_himars_random_ammo_loss_effect = {
	if = { #Если у страны ещё не приняты меры по защите от ударов по складам Химарсами
		limit = {
			NOT = {
				has_country_flag = himars_ammo_silo_protection
			}
		}

		#Выполнить потерю боеприпасов на полевом складе
		set_temp_variable = { ammo_loss = 0 }
		set_temp_variable_to_random = {
			var = ammo_loss 
			min = 10
			max = 30
			integer = yes
		}
		subtract_from_variable = {
			ammunition_variable = ammo_loss
		}

		if = {
			limit = {
				NOT = {
					has_country_flag = himars_ammo_silo_striked
				}
			}
			
			#Отображаем ивент на первый удар по полевому складу
			country_event = {
				id = generic.3
				days = 1
			}
		}
		else {
			set_country_flag = himars_ammo_silo_striked
			#Отображаем ивент на последующие удары по полевым складам
			country_event = {
				id = generic.4
				days = 1
			}
		}
		

	}
}
