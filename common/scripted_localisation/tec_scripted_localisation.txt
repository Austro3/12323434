# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes
defined_text = {
	name = dynamic_logo_text
	text = {
		trigger = {
			has_country_flag = nto_news
		}
		localization_key = nto_logo
	}
	text = {
		trigger = {
			has_country_flag = ukr_news
		}
		localization_key = ukr_logo
	}
	text = {
		trigger = {
			has_country_flag = rus_news
		}
		localization_key = rus_logo
	}
	text = {
		localization_key = neu_logo
	}
}

defined_text = {
	name = word_name
	text = {
		trigger = {
			has_global_flag = war_has_started
			has_government = neutrality
		}
		localization_key = smo_name
	}
	text = {
		trigger = {
			has_global_flag = war_has_started
			has_government = democratic
		}
		localization_key = invasion_name
	}
	text = {
		trigger = {
			has_global_flag = war_has_started
			OR = {
				has_government = communism
				has_government = fascism
			}
		}
		localization_key = war_name
	}
	text = {
		localization_key = no_name
	}
}
