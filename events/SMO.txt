#Starting the SMO (Начало СВО)
country_event = {
    id = smo.1
    title = smo.1.t
    desc = smo.1.desc
    is_triggered_only = yes
    fire_only_once = yes
    option = {
        name = smo.1.a
        declare_war_on = {
            target = UKR
            type = annex_everything
                    }
        clear_variable = global.escalation
    }
}

    id = smo.2

    is_triggered_only = yes
    fire_only_once = yes
    option = {
        name = smo.2.a

    }

    option = {
        name = smo.2.a

            value = 10
        }
    }
}

#Escalation event 1(Начало Застойки)

country_event = {
    id = smo.3
    title = smo.3.t
    desc = smo.3.desc
    fire_only_once = no 
    trigger = {
                check_variable = {
                    var = global.escalation
                    value = 50
                    compare = less_than
                }
                check_variable = {
                    var = global.escalation
                    value = 19
                    compare = greater_than
                }
    }
    option = {
        name = smo.3.a
        add_to_variable = {
            var = global.escalation
            value = 10
        } 
    }  
    option = {
        name = smo.3.b
        add_to_variable = {
            var = global.escalation
            value = 30
        }
    }   
    option = {
        name = smo.3.c
        subtract_from_variable = {
            var = global.escalation
            value = -10
        }
    }  
 }
