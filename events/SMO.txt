#Starting the SMO (Начало СВО)
country_event = {
    id = smo.1
    title = smo.1.t
    desc = smo.1.desc
    is_triggered_only = yes
    fire_only_once = yes
    option = {
        name = smo.1.a
        declare_war_on = {
            target = UKR
            type = annex_everything
                    }
            if = {
                limit = {
                    BLR = {
                        is_in_faction_with = RUS
                    }
                }
            }
            BLR = {
                add_ideas = {
                    partially_involved_in_smo_in_ukraine
                }
            }
            if = {
                limit = {
                    KAZ = {
                        is_in_faction_with = RUS
                    }
                }
            }
            KAZ = {
                add_ideas = {
                    does_not_participate_in_smo_in_ukraine
                }
            }
            set_variable = {
                var = global.escalation
                value = 0
           }
    }
}
